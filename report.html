monthly_raw <- read_csv("data/raw/74174b02-76bc-4d10-ae7c-401d90ef033c.csv")
stations_raw <- read_csv("data/raw/_published_public_transport_annual_metropolitan_train_station_entries_annual_metropolitan_train_station_entries_fy_2024_2025.csv")
monthly_clean <- monthly_raw |>
  clean_names() |>
  mutate(across(c(metropolitan_train, metropolitan_tram, metropolitan_bus,
                  regional_train, regional_coach, regional_bus),
                ~ parse_number(as.character(.)))) |>
  pivot_longer(
    cols = c(metropolitan_train, metropolitan_tram, metropolitan_bus,
             regional_train, regional_coach, regional_bus),
    names_to = "mode",
    values_to = "boardings"
  )
stations_clean <- stations_raw |>
  clean_names() |>
  filter(fin_year == "FY24-25") |>
  mutate(across(c(pax_annual, pax_weekday, pax_norm_weekday, pax_sch_hol_weekday,
                  pax_saturday, pax_sunday, pax_pre_am_peak, pax_am_peak,
                  pax_interpeak, pax_pm_peak, pax_pm_late),
                ~ parse_number(as.character(.))))
write_csv(monthly_clean, "data/processed/monthly_clean.csv")
write_csv(stations_clean, "data/processed/stations_clean.csv")
mode_summary <- monthly_clean |>
  group_by(mode) |>
  summarise(total_boardings = sum(boardings, na.rm = TRUE), .groups = "drop") |>
  arrange(desc(total_boardings))
